  function calcular() {
    const ids = ["seg", "ter", "qua", "qui", "sex", "sab", "dom"];
    const consumos = ids.map(id => parseInt(document.getElementById(id).value, 10) || 0);

    const total = consumos.reduce((soma, valor) => soma + valor, 0);
    const totalGasto = total * 5; // R$ 5 por cerveja

    const tbody = document.getElementById("tabela-body");
    tbody.innerHTML = ""; // limpa antes de adicionar

    const tr = document.createElement("tr");

    consumos.forEach(valor => {
      const td = document.createElement("td");
      td.textContent = valor;
      tr.appendChild(td);
    });

    const tdTotal = document.createElement("td");
    tdTotal.textContent = `R$ ${totalGasto.toFixed(2)}`;
    tr.appendChild(tdTotal);

    tbody.appendChild(tr);
  }

  function mostrarVideoOuImagem() {
    const midia = midias[Math.floor(Math.random() * midias.length)];
    let html = "";

    if (midia.tipo === "youtube") {
      const url = new URL(midia.url);
      const videoId = url.searchParams.get("v");
      html = `<iframe width="560" height="315" src="https://www.youtube.com/embed/${videoId}" 
      title="YouTube video player" frameborder="0" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>`;
    } else if (midia.tipo === "dailymotion") {
      const videoId = midia.url.split("/video/")[1].split('_')[0];
      html = `<iframe frameborder="0" width="560" height="315" src="https://www.dailymotion.com/embed/video/${videoId}" allowfullscreen allow="autoplay"></iframe>`;
    } else if (midia.tipo === "imagem") {
      html = `<a href="${midia.linkExterno}" target="_blank" rel="noopener noreferrer" title="Link externo">
                <img src="${midia.url}" alt="Imagem relacionada" />
              </a>`;
    } else {
      html = "<p>Tipo de mídia desconhecido</p>";
    }

    document.getElementById("video-player").innerHTML = html;
  }

  // Função para gerar string aleatória (números, letras, especiais)
  function gerarStringAleatoria(tamanho = 500) {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()-_=+[]{}|;:,.<>?';
    let resultado = '';
    for (let i = 0; i < tamanho; i++) {
      resultado += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return resultado;
  }
